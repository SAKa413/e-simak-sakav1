  `username` varchar(50) NOT NULL UNIQUE,
  `password` varchar(255) NOT NULL,
  `role` varchar(20) NOT NULL,
  `nama` varchar(100) NOT NULL,
  `unit` varchar(20) NOT NULL,

  `bio` text DEFAULT NULL,
  `alamat` text DEFAULT NULL,
